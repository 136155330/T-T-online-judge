#include <stdio.h>
int main()
{
    int a1,b1,c1,a2,b2,c2;int i;
    int n;
    scanf("%d", &n);
    for(i=0;i<n;i++)
    {
    {
        scanf("%d %d %d %d %d %d", &a1, &b1, &c1, &a2, &b2, &c2);
        a1 += a2;
        b1 += b2;
        c1 += c2;
        if(c1>=60)
        {
            c1 = c1-60;
            b1++;
        }
        if(b1>=60)
        {
            b1 = b1-60;
            a1++;
        }
        printf("%d %d %d\n", a1, b1, c1);
    }}
    return 0;
}
