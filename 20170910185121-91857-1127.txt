#include<iostream>
#include<stdio.h>
using namespace std;
int x[12];
int main()
{int y;int i;int z;int a,b,c;
    x[0]=x[2]=x[4]=x[6]=x[7]=x[9]=x[11]=31;
    x[1]=x[3]=x[5]=x[8]=x[10]=30;
    while(scanf("%d/%d/%d",&a,&b,&c)!=EOF)
    {y=0;z=0;
       for(i=0;i<b-1;i++)
       {
           y=y+x[i];
       }
       if((b>2&&a%4==0&&a%100!=0)||a%400==0)y=y-1;
      else if(b>2)y=y-2;
       z=y+c;
       cout<<z<<endl;

    }
    return 0;
}
