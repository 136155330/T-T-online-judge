#include<iostream>
using namespace std;
int x[10005];
int main()
{
    int i;int n;int y;
    while(cin>>n)
    {
        y=n/7;
        for(i=0;i<=y;i++)
        {
            x[i]=7*i;
        }
        for(i=0;i<=y;i++)
        {
            if(x[i]>10000&&(x[i]/10000)%7==0&&x[i]/10000!=0)cout<<x[i]<<endl;
            if(x[i]>10000&&(x[i]/10000)%7==0&&x[i]/10000!=0)continue;
            if(x[i]>1000&&(x[i]%10000/1000)%7==0&&(x[i]%10000/1000)!=0)cout<<x[i]<<endl;
            if(x[i]>1000&&(x[i]%10000/1000)%7==0&&(x[i]%10000/1000)!=0)continue;
            if(x[i]>100&&(x[i]%10000%1000/100)%7==0&&(x[i]%10000%1000/100)!=0)cout<<x[i]<<endl;
            if(x[i]>100&&(x[i]%10000%1000/100)%7==0&&(x[i]%10000%1000/100)!=0)continue;
            if(x[i]>10&&(x[i]%10000%1000%100/10)%7==0&&(x[i]%10000%1000%100/10)!=0)cout<<x[i]<<endl;
            if(x[i]>10&&(x[i]%10000%1000%100/10)%7==0&&(x[i]%10000%1000%100/10)!=0)continue;
            if(x[i]>0&&(x[i]%10000%1000%100%10)%7==0&&(x[i]%10000%1000%100%10)!=0)cout<<x[i]<<endl;
            if(x[i]>0&&(x[i]%10000%1000%100%10)%7==0&&(x[i]%10000%1000%100%10)!=0)continue;
        }
    }
    return 0;
}
