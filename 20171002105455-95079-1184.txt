#include<stdio.h>
#include<string.h>
int x[1000005];
int main()
{
    int t;int i;int m;int h;int max1;int l;
    while(scanf("%d",&t)!=EOF)
    {max1=0;
        memset(x,0,sizeof(0));
        for(i=0;i<t;i++)
        {
            scanf("%d",&x[i]);
        }
        for(i=0;i<t;i++)
        {
            if(x[i]>=max1){max1=x[i];l=i+1;}
        }
        while(scanf("%d",&m)!=EOF)
        {
            for(i=0;i<m;i++)
            {
                scanf("%d",&h);
               if(l+h-1<=t)printf("%d\n",x[l+h-1]);
               else
                printf("%d\n",x[l+h-1-t]);
            }
            if(i==m)break;
        }
    }
    return 0;
}
