#include<iostream>
#include<math.h>
using namespace std;
int x[1005];
int main()
{
    int n;int i;int y;int z;int j;
    while(cin>>n)
    {j=0;z=0;
        for(i=1;i<=n;i++)
        {
            x[i]=i;
            y=pow(2,j);
            if(i!=y)
            {
                x[i]=-x[i];
            }
            else
                j++;
        }
        for(i=1;i<=n;i++)
        {
            z=z+x[i];
        }
        cout<<z<<endl;
    }
    return 0;
}
