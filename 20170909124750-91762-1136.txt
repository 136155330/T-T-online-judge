#include<iostream>
#include<iomanip>
using namespace std;
int x[100];
int main()
{int a,i,j,t;int y;double z;
while(cin>>a)
{
for(i=0;i<a;i++)
{cin>>x[i];
}
for(j=a-1;j!=0;j--)
{for(i=0;i<j;i++)
{if(x[i]>x[i+1])
{t=x[i+1];
x[i+1]=x[i];
x[i]=t;
}
}
}
y=0;
for(i=1;i<=a-2;i++)
{
  y=y+x[i];
}
z=y/(a-2.0);
cout<<fixed<<setprecision(2)<<z<<endl;
}
return 0;}
