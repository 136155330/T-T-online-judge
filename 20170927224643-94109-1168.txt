#include<iostream>
using namespace std;
long long x[55];
int main()
{
    int n;int i;
    x[0]=1;
    x[1]=2;
    while(cin>>n)
    {
        if(n<=2)cout<<x[n-1]<<endl;
        else
        {
            for(i=2;i<n;i++)
            {x[i]=x[i-1]+x[i-2];}
            cout<<x[n-1]<<endl;
        }
    }
    return 0;
}
