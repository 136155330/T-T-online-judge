#include<iostream>
using namespace std;
int main()
{
    int d1,d2,d3;int x1;int x2;int x3;int x4;
    while(cin>>d1>>d2>>d3)
    {
      x1=d1+d2+d3;
      x2=d1+d1+d2+d2;
      x3=d3+d3+d2+d2;
      x4=d1+d3+d3+d1;
      if(x1<=x2&&x1<=x3&&x1<=x4)cout<<x1<<endl;
      else if(x2<=x1&&x2<=x3&&x2<=x4)cout<<x2<<endl;
      else if(x3<=x1&&x3<=x2&&x3<=x4)cout<<x3<<endl;
      else if(x4<=x1&&x4<=x2&&x4<=x3)cout<<x4<<endl;
    }
    return 0;
}
