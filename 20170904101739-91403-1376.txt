#include<iostream>
#include<stdio.h>
#include<string.h>
int n;int x[1000000];
using namespace std;
int main()
{
    int b;int i;int a;
    while(scanf("%d",&n)!=EOF)
    {memset(x,0,sizeof(x));
        for(i=0;i<n;i++)
        {
            scanf("%d",&a);
            x[a]++;
            if(x[a]>=(n+1)/2)
                b=a;
        }
        cout<<b<<endl;
    }
        return 0;
}
