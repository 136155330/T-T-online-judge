#include<iostream>
using namespace std;
int main()
{
    int n,m;int x[1000];int y;int j;int i;
    while(cin>>n>>m)
    {i=0;
        if(n<0){n=-n;cout<<"-";}
        if(n==0){cout<<'0';}
        while(n>0)
        {
            y=n%m;
            x[i]=y;
            n=n/m;
            i++;
        }
        for(j=i-1;j!=-1;j--)
        {
            if(x[j]==10)cout<<'A';
            else if(x[j]==11)cout<<'B';
            else if(x[j]==12)cout<<'C';
            else if(x[j]==13)cout<<'D';
            else if(x[j]==14)cout<<'E';
            else if(x[j]==15)cout<<'F';
            else cout<<x[j];
        }
        cout<<endl;
        }return 0;
    }
