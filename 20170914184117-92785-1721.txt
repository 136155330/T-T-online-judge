#include<iostream>
using namespace std;
int main()
{
    int n;int a,b,c,d,e,f,x;
    while(cin>>n)
    {
        a=n/100;
        b=(n%100)/50;
        c=(n%100%50)/20;
        d=(n%100%50%20)/10;
        e=(n%100%50%20%10)/5;
        f=(n%100%50%20%10%5)/1;
        x=a+b+c+d+e+f;
        cout<<x<<endl;
    }
    return 0;
}
