#include<iostream>
#include<stdlib.h>
using namespace std;
int x[105];
int main()
{int n;int i;int j;int tem;
    while(cin>>n)
    {if(n==0)break;
        for(i=0;i<n;i++)
        {
            cin>>x[i];
        }
        for(j=n-1;j!=0;j--)
        {
            for(i=0;i<j;i++)
            {
                if(abs(x[i])<abs(x[i+1]))
                {
                    tem=x[i];
                    x[i]=x[i+1];
                    x[i+1]=tem;
                }
            }
        }
        for(i=0;i<n;i++)
        {
            if(i!=(n-1))cout<<x[i]<<" ";
                else
                cout<<x[i]<<endl;
        }
    }
    return 0;
}
