#include<iostream>
using namespace std;
int x[1000005];
int main()
{
    int t;int n;int y;int sum;int i;
    while(cin>>t)
    {
        for(i=0;i<t;i++)
        {
            while(cin>>n)
            {
                for(i=0;i<n;i++)
                {
                    cin>>x[i];
                }
                y=0;
                for(i=0;i<n;i++)
                {
                    if(x[i]>y)y=x[i];
                }sum=0;
                for(i=0;i<n;i++)
                {
                    sum=sum+x[i];
                }
                if((sum-y)>=(y-1))cout<<"Yes"<<endl;
                else
                    cout<<"No"<<endl;
            }
        }
    }
    return 0;
}
