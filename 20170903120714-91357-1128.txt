#include<iostream>
using namespace std;
int main()
{
    int n;int i;int h;
    while(cin>>n)
    {int x[n];
    h=1;
        for(i=0;i<n;i++)
        {
            cin>>x[i];
        }
        for(i=0;i<n;i++)
        {
            if(x[i]%2!=0)
                h=h*x[i];
        }
    cout<<h<<endl;
    }
    return 0;
}
