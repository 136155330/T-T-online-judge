#include<stdio.h>
#include<string.h>
int main()
{ char a[200];
 int i,j,n,x[5];
 scanf("%d",&n);
 getchar();
 while(n--)
 {
  gets(a);
  j=strlen(a);
  for(i=0;i<5;i++)
  x[i]=0;
  for(i=0;i<j;i++)
  { if(a[i]=='a')
    x[0]++;
   if(a[i]=='e')
   x[1]++;
   if(a[i]=='i')
    x[2]++;
   if(a[i]=='o')
    x[3]++;
   if(a[i]=='u')
    x[4]++;
  }
  printf("a:%d\ne:%d\ni:%d\no:%d\nu:%d\n",x[0],x[1],x[2],x[3],x[4]);
  if(n!=0)
  printf("\n");
 }
 return 0;
}
