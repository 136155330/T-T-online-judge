#include<iostream>
using namespace std;
int main()
{
    int n;int i;int a;int b;int c;
    while(cin>>n)
    {double x[n];
    a=0;b=0;c=0;
    if(n==0)break;
    for(i=0;i<n;i++)
    {
        cin>>x[i];
    }
    for(i=0;i<n;i++)
    {
        if(x[i]<0)a=a+1;
        else
        if(x[i]==0)b=b+1;
        else
        if(x[i]>0)c=c+1;
    }
    cout<<a<<" "<<b<<" "<<c<<endl;
    }
    return 0;
}
