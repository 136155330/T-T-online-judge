#include<iostream>
using namespace std;
int x[100];int y[100];
int main()
{
    int n;int tem;int z;int a;int i;
    while(cin>>n)
    {
        if(n==0)break;
        for(i=0;i<n;i++)
        {
            cin>>x[i];
        }
        for(i=0;i<n;i++)
        {
            y[i]=x[i];
        }
        for(i=0;i<n;i++)
        {
           if(x[i]<x[i+1])
           {tem=x[i+1];
               x[i+1]=x[i];
           x[i]=tem;
           z=x[i+1];
           }
        }
        for(i=0;i<n;i++)
        {
            if(y[i]==z)
            {
                a=y[0];
                y[0]=z;
                y[i]=a;
            }
        }
        for(i=0;i<n;i++)
        {if(i!=(n-1))
            cout<<y[i]<<" ";
            else
                cout<<y[i]<<endl;

        }




    }
    return 0;
}
