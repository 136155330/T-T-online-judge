#include<string.h>
#include<iostream>
#include<stdio.h>
using namespace std;
int x[105];
int main()
{
    int n;int i;int y;int j;int z;
    while(scanf("%d",&n)!=EOF)
    {memset(x,0,sizeof(x));
        for(i=0;i<n;i++)
        {
            scanf("%d",&y);
            x[y]++;
        }
        z=0;
        for(i=100;i!=-1;i--)
        {
            if(x[i]!=0&&x[i]==1){if(z!=n-1){cout<<i<<" ";z++;}
            else
                cout<<i<<endl;
            }
            if(x[i]!=0&&x[i]>1)
            {
                for(j=x[i];j!=0;j--){if(z!=n-1){cout<<i<<" ";z++;}
                else
                    cout<<i<<endl;
                }
            }
        }
    }
    return 0;
}
