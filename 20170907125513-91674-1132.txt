#include<iostream>
#include<string.h>
using namespace std;
int x[1000];int y[1000];
int main()
{int m;int n;int i;int j;int a;int b;int c;int h;
while(cin>>m>>n)

{memset(x,0,sizeof(x));
memset(y,0,sizeof(y));
h=0;
    for(i=m;i<=n;i++)
    {
        a=i/100;
        b=(i%100)/10;
        c=i%10;
        if(i==(a*a*a)+(b*b*b)+(c*c*c))x[i]++;
    }
    for(i=m;i<=n;i++)
    {
        if(x[i]!=0){y[h]=i;
        h++;}
    }
   for(i=0;i<=(h-1);i++)
   {
       if(y[i]!=0&&i!=(h-1))cout<<y[i]<<" ";
        else if(y[i]!=0&&i==(h-1)) cout<<y[i];
   }
    for(i=m;i<=n;i++)
    {
        j=x[i];
        if(j!=0)cout<<endl;
        if(j!=0)break;
    }
    if(j==0)cout<<"no"<<endl;
    }


    return 0;
}
