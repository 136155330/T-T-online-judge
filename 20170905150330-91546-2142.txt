#include<iostream>
#include<string.h>
using namespace std;int x[10000];
int main()

{
    int n;int m;int j;int i;
    while(cin>>n)
    {j=0;
        memset(x,0,sizeof(x));
        for(i=0;i<n;i++)
        {
            cin>>m;
            x[m]++;
        }
        for(i=0;i<=n;i++)
        {
            if(x[i]>0)j++;
        }
        cout<<j<<endl;
    }
    return 0;
}
