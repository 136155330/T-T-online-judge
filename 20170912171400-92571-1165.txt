#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
char x[55];
int main()
{
    int n;int a,b,c,d;int y;int i;int j;
    while(cin>>n)
    {getchar();
        for(i=0;i<n;i++)
        {a=b=c=d=0;y=0;
        memset(x,0,sizeof(x));
            gets(x);
            y=strlen(x);
            for(j=0;j<y;j++)
            {
                if(x[j]>='a'&&x[j]<='z')a=1;
                if(x[j]>='A'&&x[j]<='Z')b=1;
                if(x[j]>='0'&&x[j]<='9')c=1;
                if(x[j]=='~'||x[j]=='!'||x[j]=='@'||x[j]=='#'||x[j]=='$'||x[j]=='%'||x[j]=='^')d=1;

            }
            if(y>=8&&y<=16&&((a+b+c+d)>=3))cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;
        }
    }
    return 0;
}
